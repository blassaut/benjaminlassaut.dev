# GasLens Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone multi-chain EVM gas tracker that shows real-time gas prices across 8 chains in a single dashboard.

**Architecture:** Supabase Edge Function polls gas prices from public RPCs every ~15s (4 cycles per 1-min cron), stores in Postgres. React frontend fetches latest prices on mount, then subscribes to Supabase Realtime for live updates. Deployed on Vercel.

**Tech Stack:** React 19, Vite 7, TypeScript, Tailwind CSS v4, Supabase (Postgres + Edge Functions + Realtime + pg_cron), Deno (Edge Function runtime)

**Spec:** `docs/superpowers/specs/2026-03-17-gaslens-design.md`

```typescript
export function formatGwei(gwei: number): string {
  if (gwei >= 1) return gwei.toFixed(1)
  if (gwei >= 0.01) return gwei.toFixed(3)
  return gwei.toFixed(4)
}

export function formatUsd(usd: number): string {
  if (usd < 0.01) return '<$0.01'
  return `$${usd.toFixed(2)}`
}

export type GasLevel = 'green' | 'yellow' | 'red'

export function getGasLevel(gwei: number, thresholds: { green: number; yellow: number }): GasLevel {
  if (gwei < thresholds.green) return 'green'
  if (gwei < thresholds.yellow) return 'yellow'
  return 'red'
}
```

- [ ] **Step 2: Create the `useGasPrices` hook**

Create `src/hooks/useGasPrices.ts`:

```typescript
import { useEffect, useState } from 'react'
import { supabase } from '../lib/supabase'
import { GasPrice } from '../lib/types'
import { chains } from '../lib/chains'

export function useGasPrices() {
  const [prices, setPrices] = useState<Map<number, GasPrice>>(new Map())
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    async function fetchLatest() {
      // Supabase JS doesn't support DISTINCT ON directly.
      // Fetch the most recent rows and deduplicate by chain_id client-side.
      const { data: recentData, error: recentError } = await supabase
        .from('gas_prices')
        .select('*')
        .in('chain_id', chains.map((c) => c.id))
        .order('fetched_at', { ascending: false })
        .limit(80) // 8 chains x 10 rows buffer

      if (recentError) {
        setError(recentError.message)
        setLoading(false)
        return
      }

      const latestByChain = new Map<number, GasPrice>()
      for (const row of recentData as GasPrice[]) {
        if (!latestByChain.has(row.chain_id)) {
          latestByChain.set(row.chain_id, row)
        }
      }

      setPrices(latestByChain)
      setLoading(false)
    }

    fetchLatest()

    // Subscribe to Realtime INSERT events
    const channel = supabase
      .channel('gas_prices_realtime')
      .on(
        'postgres_changes',
        { event: 'INSERT', schema: 'public', table: 'gas_prices' },
        (payload) => {
          const newPrice = payload.new as GasPrice
          setPrices((prev) => {
            const next = new Map(prev)
            next.set(newPrice.chain_id, newPrice)
            return next
          })
        }
      )
      .subscribe()

    return () => {
      supabase.removeChannel(channel)
    }
  }, [])

  return { prices, loading, error }
}
```

Create `supabase/migrations/00001_create_gas_prices.sql`:

```sql
create table gas_prices (
  id uuid primary key default gen_random_uuid(),
  chain_id int not null,
  chain_name text not null,
  gas_low numeric not null,
  gas_medium numeric not null,
  gas_high numeric not null,
  base_fee numeric not null,
  usd_cost_medium numeric not null,
  fetched_at timestamptz not null default now()
);

create index idx_gas_prices_chain_time on gas_prices (chain_id, fetched_at desc);

-- Enable RLS
alter table gas_prices enable row level security;

-- Anon can read
create policy "gas_prices_select" on gas_prices
  for select to anon using (true);

-- Only service role can insert
create policy "gas_prices_insert" on gas_prices
  for insert to service_role with check (true);

-- Enable Realtime
alter publication supabase_realtime add table gas_prices;
```

```sql
-- Compaction function: aggregate granular data older than 24h into 5-minute buckets
create or replace function compact_gas_prices()
returns void
language plpgsql
as $$
declare
  cutoff timestamptz := now() - interval '24 hours';
begin
  -- Insert 5-minute averages for data older than 24 hours
  insert into gas_prices_5m (chain_id, chain_name, gas_low_avg, gas_medium_avg, gas_high_avg, base_fee_avg, usd_cost_medium_avg, bucket)
  select
    chain_id,
    chain_name,
    avg(gas_low),
    avg(gas_medium),
    avg(gas_high),
    avg(base_fee),
    avg(usd_cost_medium),
    date_trunc('hour', fetched_at) + (extract(minute from fetched_at)::int / 5) * interval '5 minutes'
  from gas_prices
  where fetched_at < cutoff
  group by chain_id, chain_name, date_trunc('hour', fetched_at) + (extract(minute from fetched_at)::int / 5) * interval '5 minutes';

  -- Delete compacted rows using same cutoff to avoid race condition
  delete from gas_prices
  where fetched_at < cutoff;
end;
$$;

-- Schedule compaction every 5 minutes via pg_cron
-- (Run this manually in Supabase SQL Editor if pg_cron is not available in migrations)
select cron.schedule(
  'compact-gas-prices',
  '*/5 * * * *',
  'select compact_gas_prices()'
);
```

```typescript
  { id: 324, name: 'zkSync', primaryRpc: Deno.env.get('RPC_ZKSYNC') || 'https://mainnet.era.zksync.io', fallbackRpc: 'https://rpc.ankr.com/zksync_era', nativeToken: 'ethereum' },

async function rpcCall(url: string, method: string, params: unknown[] = []): Promise<unknown> {
  const res = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ jsonrpc: '2.0', id: 1, method, params }),
  })
  const json = await res.json()
  if (json.error) throw new Error(json.error.message)
  return json.result
}

async function fetchGasForChain(chain: ChainConfig): Promise<{ low: number; medium: number; high: number; baseFee: number } | null> {
  for (const rpcUrl of [chain.primaryRpc, chain.fallbackRpc]) {
    try {
      // Get fee history for last 20 blocks
      const feeHistory = await rpcCall(rpcUrl, 'eth_feeHistory', ['0x14', 'latest', [10, 50, 90]]) as {
        baseFeePerGas: string[]
        reward: string[][]
      }

      const latestBaseFee = parseInt(feeHistory.baseFeePerGas[feeHistory.baseFeePerGas.length - 1], 16) / 1e9

      // Extract priority fee percentiles and average them
      const rewards = feeHistory.reward || []
      let lowSum = 0, medSum = 0, highSum = 0
      for (const block of rewards) {
        lowSum += parseInt(block[0], 16) / 1e9
        medSum += parseInt(block[1], 16) / 1e9
        highSum += parseInt(block[2], 16) / 1e9
      }
      const count = rewards.length || 1

      return {
        low: latestBaseFee + lowSum / count,
        medium: latestBaseFee + medSum / count,
        high: latestBaseFee + highSum / count,
        baseFee: latestBaseFee,
      }
    } catch (err) {
      console.error(`RPC failed for ${chain.name} at ${rpcUrl}:`, err)
      continue
    }
  }
  return null
}

async function fetchTokenPrices(): Promise<Map<string, number>> {
  const tokenIds = [...new Set(chains.map((c) => c.nativeToken))].join(',')
  try {
    const res = await fetch(`https://api.coingecko.com/api/v3/simple/price?ids=${tokenIds}&vs_currencies=usd`)
    const json = await res.json()
    const prices = new Map<string, number>()
    for (const [id, data] of Object.entries(json)) {
      prices.set(id, (data as { usd: number }).usd)
    }
    return prices
  } catch (err) {
    console.error('CoinGecko fetch failed:', err)
    return new Map()
  }
}

async function pollCycle(tokenPrices: Map<string, number>) {
  const results = await Promise.allSettled(
    chains.map(async (chain) => {
      const gas = await fetchGasForChain(chain)
      if (!gas) return null

      const nativePrice = tokenPrices.get(chain.nativeToken) || 0
      // Standard transfer = 21000 gas. Cost = gasPrice (gwei) * 21000 / 1e9 * nativeTokenUSD
      const usdCost = gas.medium * 21000 / 1e9 * nativePrice

      return {
        chain_id: chain.id,
        chain_name: chain.name,
        gas_low: gas.low,
        gas_medium: gas.medium,
        gas_high: gas.high,
        base_fee: gas.baseFee,
        usd_cost_medium: usdCost,
        fetched_at: new Date().toISOString(),
      }
    })
  )

  const rows = results
    .filter((r) => r.status === 'fulfilled' && r.value !== null)
    .map((r) => (r as PromiseFulfilledResult<NonNullable<unknown>>).value)

  if (rows.length > 0) {
    const { error } = await supabase.from('gas_prices').insert(rows)
    if (error) console.error('Insert error:', error)
  }
}

function sleep(ms: number) {
  return new Promise((resolve) => setTimeout(resolve, ms))
}

Deno.serve(async () => {
  // Fetch token prices once per invocation (cached across all 4 cycles)
  const tokenPrices = await fetchTokenPrices()

  // Run 4 cycles with ~15s intervals
  for (let i = 0; i < 4; i++) {
    await pollCycle(tokenPrices)
    if (i < 3) await sleep(15000)
  }

  return new Response(JSON.stringify({ ok: true, cycles: 4 }), {
    headers: { 'Content-Type': 'application/json' },
  })
})
```
